.PHONY: venv install dev test lint clean clean-demo-readmes demo-readmes

venv:
	python3 -m venv .venv

install: venv
	# Ensure the venv's bin directory is first in PATH for these commands
	PATH=.venv/bin:$$PATH .venv/bin/pip install --upgrade pip
	PATH=.venv/bin:$$PATH .venv/bin/pip install -e .[dev]

dev: install
	PATH=.venv/bin:$$PATH .venv/bin/pre-commit install
	@echo "Development environment installed."
	@echo "To use the virtualenv in your shell, run: . .venv/bin/activate"
	@echo "Or prepend it to PATH for one-off commands: export PATH=.venv/bin:$$PATH"

test:
	. .venv/bin/activate && pytest -q

lint:
	. .venv/bin/activate && tox -e lint,pylint

clean-demo-readmes:
	rm -f REVIEW_README.md
	if [ -d debug_readmes ]; then find debug_readmes -mindepth 1 -delete; fi
	mkdir -p debug_readmes

demo-readmes:
	set -e; \
	BASE_ROLE="src/prism/tests/roles/base_mock_role"; \
	INROLE_CFG_ROLE="src/prism/tests/roles/inrole_config_role"; \
	python -m prism.cli role "$$BASE_ROLE" -o debug_readmes/REVIEW_README.md -v; \
	python -m prism.cli role "$$BASE_ROLE" -o debug_readmes/REVIEW_README.html -f html -v; \
	python -m prism.cli role "$$BASE_ROLE" -o debug_readmes/REVIEW_README.json -f json -v; \
	python -m prism.cli role "$$BASE_ROLE" -o debug_readmes/REVIEW_README_CONCISE.md --concise-readme --scanner-report-output debug_readmes/SCAN_REPORT.md -v; \
	python -m prism.cli role "$$BASE_ROLE" -o debug_readmes/REVIEW_README_STYLE_GUIDE_SKELETON.md --create-style-guide -v; \
	python -m prism.cli role "$$INROLE_CFG_ROLE" -o debug_readmes/REVIEW_README_INROLE_CONFIG.md -v

clean:
	rm -rf debug_readmes
	rm -rf .venv build dist *.egg-info
