.PHONY: venv install dev test lint clean clean-demo-readmes

venv:
	python3 -m venv .venv

install: venv
	# Ensure the venv's bin directory is first in PATH for these commands
	PATH=.venv/bin:$$PATH .venv/bin/pip install --upgrade pip
	PATH=.venv/bin:$$PATH .venv/bin/pip install -e .[dev]

dev: install
	# Dev target finishes by reminding how to activate the venv.
	@echo "Development environment installed."
	@echo "To use the virtualenv in your shell, run: . .venv/bin/activate"
	@echo "Or prepend it to PATH for one-off commands: export PATH=.venv/bin:$$PATH"

test:
	. .venv/bin/activate && pytest -q

lint:
	. .venv/bin/activate && tox -e lint,pylint

clean-demo-readmes:
	rm -f REVIEW_README.md
	if [ -d debug_readmes ]; then find debug_readmes -mindepth 1 -delete; fi
	mkdir -p debug_readmes

clean:
	rm -rf debug_readmes
	rm -rf .venv build dist *.egg-info
